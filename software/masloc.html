Hello world with Linux syscalls

// obviously, much of this could be inlined and in a real program should not be // uselessly extracted into their own functions. // this implementation is simply meant to make `_start` as similar as possible // to the version that uses the standard library. fn [abi := "sysv"] exit (rc: i32) { asm { "\tmov $60, %rax\n" "\tsyscall\n" } } fn [abi := "sysv"] write (fd: u32, s: u8*, n: u64) { asm { "\tmov $1, %rax\n" "\tsyscall\n" } } fn strlen: u64 <- (s: u8*) { dim n: u64 := 0; while 0 n s + [] != { n +@; } ret n; } fn puts (s: u8*) { ((s strlen) s 1 write); } glob fn _start { ("hello world\n" puts); (0 exit); }
